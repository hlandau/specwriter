% Make chapter/section headings have the section number in the margin.
\unexpanded\def\PlaceSection#1#2%
  {\goodbreak
   \vbox
     {\localheadsetup
      \begstrut
      \inleftmargin{ #1}%
      #2}}

\setupheads[]
\setuphead[chapter][page=yes,before={\blank[10*big,force]},after=,style=\bfc,command=\PlaceSection]
